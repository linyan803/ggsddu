select english_words.ID, SUBJECT, TYPE, TIMES, CORRECT, STATUS, WEIGHT, DES,WORD from english_words join exercise_info where (en)==exercise_info.ID
order by weight,english_words.ID DESC limit 20;

CREATE TABLE exercise_info (
SUBJECT INT8 NOT NULL,
ID INT NOT NULL, 
ITEM_NUM INT8,
TIMES INT,
CORRECT INT,
STATUS INT8,
NOTE VARCHAR(255), 
WEIGHT REAL,
CONSTRAINT PK PRIMARY KEY (SUBJECT,ID));


CREATE TABLE update_info (
ID INT PRIMARY KEY NOT NULL,
UPDATE_DATE DATETIME, 
NOTE VARCHAR(255));

CREATE TABLE stem (
ID INT PRIMARY KEY NOT NULL,
MODEL INT8, 
DES_STYLE INT8,
DES TEXT);

CREATE TABLE key (
ID INT PRIMARY KEY NOT NULL,
SUB_ID INT8,
KEY  VARCHAR(255));

CREATE TABLE answer (
ID INT PRIMARY KEY NOT NULL,
SUB_ID INT8,
ANSWER  VARCHAR(8));
